   export function useCatalogItems(pageSize: number = 10, pageIndex: number = 0) {
     return useQuery<PaginatedCatalogItemsResponse>({
       queryKey: ['catalog-items', pageSize, pageIndex],
       queryFn: () => getCatalogItems(pageSize, pageIndex),
       staleTime: 1000 * 60,

     })
